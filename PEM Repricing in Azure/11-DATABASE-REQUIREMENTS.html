Database Requirements for Containerized Repricing

Version: 1.0
Last Updated: January 16, 2026
Author: Andy Bratton
Status: Deep-Dive Technical Reference

Overview

This document catalogs all database tables, reference data, and data access patterns required for the containerized repricing engine. Understanding these dependencies is critical for determining: 1. What data must be accessible from the container 2. What data can be embedded vs. queried at runtime 3. What caching strategies to employ

1. Database Architecture

Current Production Environment

Database Hierarchy:

Primary SQL Server CBO_Global 235 GB

10. Data Access Layer Design

Embedded Data Loader

// ReferenceDataLoader.cs
public class ReferenceDataLoader
{
    public async Task LoadEmbeddedDataAsync(string dataPath)
    {
        // Load from JSON files embedded in container
        _drgWeights = await LoadJsonAsync<Dictionary<string, DrgWeight>>(
            Path.Combine(dataPath, "drg-weights.json"));
        
        _apcRates = await LoadJsonAsync<Dictionary<string, ApcRate>>(
            Path.Combine(dataPath, "apc-rates.json"));
        
        _rvuTable = await LoadJsonAsync<Dictionary<string, RvuData>>(
            Path.Combine(dataPath, "rvu-table.json"));
        
        _gpciTable = await LoadJsonAsync<Dictionary<string, GpciData>>(
            Path.Combine(dataPath, "gpci-table.json"));
    }
    
    public DrgWeight GetDrgWeight(string drgCode, string version)
    {
        var key = $"{drgCode}|{version}";
        return _drgWeights.TryGetValue(key, out var weight) ? weight : null;
    }
}

Cached Data Manager

// CachedDataManager.cs
public class CachedDataManager
{
    private readonly IMemoryCache _cache;
    private readonly ISqlConnection _db;
    
    public async Task<ProviderRates> GetProviderRatesAsync(
        string providerId, 
        int fiscalYear)
    {
        var key = $"rates:{providerId}:{fiscalYear}";
        
        if (_cache.TryGetValue(key, out ProviderRates rates))
            return rates;
        
        // Query database
        rates = await _db.QueryFirstOrDefaultAsync<ProviderRates>(
            "SELECT * FROM MCRIPHSPRATES WHERE PROVIDER_ID = @ProviderId AND FISCAL_YEAR = @FiscalYear",
            new { ProviderId = providerId, FiscalYear = fiscalYear });
        
        // Cache for 24 hours
        _cache.Set(key, rates, TimeSpan.FromHours(24));
        
        return rates;
    }
}

11. Data Synchronization

Embedded Data Build Pipeline

# azure-pipelines-data.yml
trigger:
  paths:
    include:
      - data-exports/**

stages:
  - stage: ExportData
    jobs:
      - job: ExportReferenceData
        steps:
          - task: SqlAzure@3
            inputs:
              scriptPath: 'scripts/export-drg-weights.sql'
              outputPath: '$(Build.ArtifactStagingDirectory)/drg-weights.json'
          
          - task: SqlAzure@3
            inputs:
              scriptPath: 'scripts/export-apc-rates.sql'
              outputPath: '$(Build.ArtifactStagingDirectory)/apc-rates.json'
          
          - task: SqlAzure@3
            inputs:
              scriptPath: 'scripts/export-rvu-table.sql'
              outputPath: '$(Build.ArtifactStagingDirectory)/rvu-table.json'
          
          - publish: '$(Build.ArtifactStagingDirectory)'
            artifact: 'reference-data'

  - stage: BuildContainer
    dependsOn: ExportData
    jobs:
      - job: BuildPricingWorker
        steps:
          - download: current
            artifact: 'reference-data'
          
          - task: Docker@2
            inputs:
              command: 'build'
              Dockerfile: 'docker/Dockerfile.pricing-worker'
              buildContext: '.'
              arguments: '--build-arg DATA_PATH=$(Pipeline.Workspace)/reference-data'
