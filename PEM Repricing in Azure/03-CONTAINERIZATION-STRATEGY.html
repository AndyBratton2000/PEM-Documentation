Overview

Phase 1 packages the existing Contract Manager repricing engine into Windows containers for deployment in Azure. This approach provides:

  • ✅ Minimal Risk - Uses proven, tested pricing logic
  • ✅ Fast Time-to-Market - 6-8 weeks vs 6+ months for rewrite
  • ✅ Full Parity - Same DLLs = same results
  • ✅ Horizontal Scaling - Container orchestration enables scale-out

// NativePricingBridge.cs
public class NativePricingBridge : IPricingBridge, IDisposable
{
    private const string BridgeDll = "CboPricingBridge.dll";
    
    [DllImport(BridgeDll, CharSet = CharSet.Unicode, CallingConvention = CallingConvention.Cdecl)]
    private static extern int PriceEpisodeFromJson(
        [MarshalAs(UnmanagedType.LPWStr)] string episodeJson,
        out IntPtr resultJson,
        out uint resultLength
    );
    
    [DllImport(BridgeDll, CallingConvention = CallingConvention.Cdecl)]
    private static extern void FreePricingResult(IntPtr resultJson);
    
    [DllImport(BridgeDll, CallingConvention = CallingConvention.Cdecl)]
    private static extern int InitializePricingEngine(
        [MarshalAs(UnmanagedType.LPWStr)] string configPath
    );
    
    public async Task<PricingResult> PriceClaimAsync(ClaimInput claim)
    {
        var json = JsonSerializer.Serialize(claim);
        
        var hr = PriceEpisodeFromJson(json, out IntPtr resultPtr, out uint length);
        
        if (hr != 0)
        {
            throw new PricingException($"Native pricing failed: HRESULT=0x{hr:X8}");
        }
        
        try
        {
            var resultJson = Marshal.PtrToStringUni(resultPtr, (int)length);
            return JsonSerializer.Deserialize<PricingResult>(resultJson);
        }
        finally
        {
            FreePricingResult(resultPtr);
        }
    }
}

Implementation Strategy:

// CboPricingBridge.cpp
#include "CboPricingBridge.h"
#include "EpisodeSerializer.h"
#include "MinimalPricer.h"
#include <nlohmann/json.hpp>

// Global pricer instance
static std::unique_ptr<MinimalPricer> g_pricer;

HRESULT InitializePricingEngine(LPCWSTR configPath)
{
    try
    {
        auto config = LoadConfig(configPath);
        g_pricer = std::make_unique<MinimalPricer>(config);
        g_pricer->Initialize();
        return S_OK;
    }
    catch (const std::exception& ex)
    {
        LogError("InitializePricingEngine failed: %s", ex.what());
        return E_FAIL;
    }
}

HRESULT PriceEpisodeFromJson(
    LPCWSTR episodeJson,
    LPWSTR* resultJson,
    DWORD* resultLength)
{
    if (!g_pricer)
        return E_NOT_VALID_STATE;
    
    try
    {
        // Parse JSON to HEpisode
        auto episode = EpisodeSerializer::FromJson(episodeJson);
        
        // Determine program (Medicare IP, OP, Medicaid, etc.)
        auto program = g_pricer->DetermineProgram(episode);
        
        // Execute grouping if needed
        if (program.RequiresDRG())
        {
            g_pricer->GroupToDRG(episode);
        }
        else if (program.RequiresAPC())
        {
            g_pricer->GroupToAPC(episode);
        }
        
        // Execute pricing
        auto result = g_pricer->CalculatePayment(episode, program);
        
        // Serialize result to JSON
        auto json = EpisodeSerializer::ToResultJson(result);
        
        // Allocate output buffer
        *resultLength = (DWORD)(json.length() + 1);
        *resultJson = (LPWSTR)CoTaskMemAlloc(*resultLength * sizeof(WCHAR));
        wcscpy_s(*resultJson, *resultLength, json.c_str());
        
        return S_OK;
    }
    catch (const std::exception& ex)
    {
        LogError("PriceEpisodeFromJson failed: %s", ex.what());
        return E_FAIL;
    }
}

void FreePricingResult(LPWSTR resultJson)
{
    CoTaskMemFree(resultJson);
}

Scaling Rules

Metric Threshold Action
Queue Messages > 100 Scale out (+1 replica)
Queue Messages > 500 Scale out (+3 replicas)
Queue Messages < 10 for 5min Scale in (-1 replica)
CPU Usage > 80% Scale out
Min Replicas (Prod) 3 Always running
Max Replicas (Prod) 20 Cost control

Build Pipeline

GitHub Actions

# .github/workflows/pricing-worker.yml
name: Build Pricing Worker

on:
  push:
    branches: [main]
    paths:
      - 'src/PricingWorker/**'
      - 'src/CboPricingBridge/**'
  pull_request:
    branches: [main]

jobs:
  build-native:
    runs-on: windows-latest
    steps:
      - uses: actions/checkout@v4
      
      - name: Setup MSBuild
        uses: microsoft/setup-msbuild@v2
        
      - name: Restore NuGet packages
        run: nuget restore src/CboPricingBridge/CboPricingBridge.sln
        
      - name: Build native bridge
        run: msbuild src/CboPricingBridge/CboPricingBridge.vcxproj /p:Configuration=Release /p:Platform=x86
        
      - name: Upload native artifacts
        uses: actions/upload-artifact@v4
        with:
          name: native-bridge
          path: src/x86/Release/
          
  build-worker:
    runs-on: windows-latest
    needs: build-native
    steps:
      - uses: actions/checkout@v4
      
      - name: Setup .NET
        uses: actions/setup-dotnet@v4
        with:
          dotnet-version: '10.0.x'
          
      - name: Download native artifacts
        uses: actions/download-artifact@v4
        with:
          name: native-bridge
          path: artifacts/native/
          
      - name: Build worker
        run: dotnet publish src/PricingWorker -c Release -o publish/
        
      - name: Run tests
        run: dotnet test tests/PricingWorker.Tests
        
  build-container:
    runs-on: windows-latest
    needs: [build-native, build-worker]
    steps:
      - uses: actions/checkout@v4
      
      - name: Download all artifacts
        uses: actions/download-artifact@v4
        
      - name: Login to ACR
        uses: azure/docker-login@v1
        with:
          login-server: ${{ secrets.ACR_LOGIN_SERVER }}
          username: ${{ secrets.ACR_USERNAME }}
          password: ${{ secrets.ACR_PASSWORD }}
          
      - name: Build and push image
        run: |
          docker build -t ${{ secrets.ACR_LOGIN_SERVER }}/pricing-worker:${{ github.sha }} .
          docker push ${{ secrets.ACR_LOGIN_SERVER }}/pricing-worker:${{ github.sha }}
